to fern :size :sign

  fd :size
  rt 70 * :sign fern :size * 0.5 :sign * (-1) lt 70 * :sign
  fd :size
  lt 70 * :sign fern :size * 0.5 :sign rt 70 * :sign
  rt 7 * :sign fern :size - 1 :sign lt 7 * :sign
  bk :size * 2
end
